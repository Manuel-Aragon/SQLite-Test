CREATE TABLE IF NOT EXISTS characterTable(
	ID INTEGER PRIMARY KEY AUTOINCREMENT,
	NAME TEXT NOT NULL,
	Race TEXT NOT NULL,
	MaxHitpoints INT NOT NULL,
	CurrentHitpoints INT NOT NULL,
	ArmorClass INT NOT NULL,
	MovementSpeed INT NOT NULL);
CREATE TABLE IF NOT EXISTS playerTable(
	ID INTEGER PRIMARY KEY AUTOINCREMENT,
	NAME TEXT NOT NULL,
	Race TEXT NOT NULL,
	MaxHitpoints INT NOT NULL,
	CurrentHitpoints INT NOT NULL,
	ArmorClass INT NOT NULL,
	MovementSpeed INT NOT NULL);
